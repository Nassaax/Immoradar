<script>
'use strict';
let sb=null,page=1,total=0,sort='newest',favOnly=false,searchTimer=null;
let favs=[];
const LIMIT=18;
const f={search:'',status:'all',type:'all',city:'',agencyId:'',minPrice:null,maxPrice:null,isNew:false};
const TYPE_LABELS={house:'🏠 Maison',apartment:'🏢 Appartement',building:'🏦 Immeuble',land:'🌿 Terrain',commercial:'🏪 Commerce',other:'📦 Autre'};

(async function init(){
  try{favs=JSON.parse(localStorage.getItem('ir-favs')||'[]');}catch{}
  updateFavUI();
  try{
    const res=await fetch('/api/config');
    if(!res.ok)throw new Error('HTTP '+res.status);
    const cfg=await res.json();
    if(!cfg.ok)throw new Error(cfg.message||'Config manquante');
    sb=supabase.createClient(cfg.url,cfg.anon);
    hideStatus();loadListings();
  }catch(e){
    showStatus('setup','⚙️  Configurez SUPABASE_URL et SUPABASE_ANON dans Vercel → Settings → Environment Variables, puis cliquez Redeploy.');
    renderSetup();
  }
})();

async function loadListings(){
  renderSkeletons();
  if(!sb)return;
  try{
    let q=sb.from('listings').select('*',{count:'exact'});
    if(f.search)q=q.or(`title.ilike.%${f.search}%,description.ilike.%${f.search}%,address.ilike.%${f.search}%,city.ilike.%${f.search}%`);
    if(f.status!=='all')q=q.eq('status',f.status);
    if(f.type!=='all')q=q.eq('type',f.type);
    if(f.city)q=q.ilike('city',`%${f.city}%`);
    if(f.agencyId)q=q.eq('agency_id',f.agencyId);
    if(f.minPrice!=null)q=q.gte('price',f.minPrice);
    if(f.maxPrice!=null)q=q.lte('price',f.maxPrice);
    if(f.isNew)q=q.eq('is_new',true);
    if(favOnly&&favs.length>0)q=q.in('id',favs);
    if(favOnly&&favs.length===0){renderEmpty('favs');updateResultsInfo(0);return;}
    if(sort==='price_asc')q=q.order('price',{ascending:true,nullsFirst:false});
    else if(sort==='price_desc')q=q.order('price',{ascending:false,nullsFirst:false});
    else q=q.order('created_at',{ascending:false});
    const from=(page-1)*LIMIT;
    q=q.range(from,from+LIMIT-1);
    const{data,count,error}=await q;
    if(error)throw error;
    total=count||0;
    document.getElementById('liveCount').textContent=total.toLocaleString('fr-BE')+' annonces';
    renderListings(data||[]);
    renderPagination(total);
    updateResultsInfo(total);
  }catch(err){console.error(err);renderError(err.message);}
}

function timeAgo(iso){const h=Math.floor((Date.now()-new Date(iso))/3600000);if(h<1)return'< 1h';if(h<24)return h+'h';const d=Math.floor(h/24);return d===1?'hier':d+'j';}

function renderPagination(count){
  const tp=Math.ceil(count/LIMIT);
  const pag=document.getElementById('pagination');
  if(tp<=1){pag.style.display='none';return;}
  pag.style.display='flex';
  document.getElementById('prevBtn').disabled=page===1;
  document.getElementById('nextBtn').disabled=page===tp;
  let html='';
  const show=Math.min(tp,7);
  for(let i=1;i<=show;i++)html+=`<button class="page-dot ${i===page?'active':''}" onclick="goPage(${i})">${i}</button>`;
  if(tp>7)html+=`<span style="color:var(--text3);font-size:12px;padding:0 2px">…${tp}</span>`;
  document.getElementById('pageDots').innerHTML=html;
}

async function submitSubscribe(){
  const email=(document.getElementById('subEmail').value||'').trim();
  if(!email||!email.includes('@')){toast('⚠️ Email invalide');return;}
  const btn=document.getElementById('submitBtn');btn.disabled=true;btn.textContent='Inscription…';
  if(!sb){setTimeout(()=>{document.getElementById('modalForm').style.display='none';document.getElementById('modalSuccess').style.display='block';},600);return;}
  try{
    const filters={};
    if(f.status!=='all')filters.status=f.status;if(f.type!=='all')filters.type=f.type;
    if(f.city)filters.city=f.city;if(f.maxPrice)filters.maxPrice=f.maxPrice;
    const{error}=await sb.from('subscribers').upsert({email:email.toLowerCase(),filters,active:true},{onConflict:'email'});
    if(error)throw error;
    document.getElementById('modalForm').style.display='none';document.getElementById('modalSuccess').style.display='block';
  }catch(err){toast('❌ Erreur : '+err.message);btn.disabled=false;btn.textContent="S'inscrire gratuitement";}
}
