function getSupabase() {
const url = process.env.SUPABASE_URL;
const key = process.env.SUPABASE_SERVICE_KEY;
if (!url || !key) throw new Error(‘Variables SUPABASE_URL et SUPABASE_SERVICE_KEY manquantes dans Vercel.’);
return createClient(url, key);
}

// ── Utilitaires ───────────────────────────────────────────────────

function hash(str) {
return crypto.createHash(‘md5’).update(str.trim().toLowerCase()).digest(‘hex’);
}
function extrairePrix(texte) {
if (!texte) return { prix:null, prix_texte:null };
const t = texte.trim().replace(/\s+/g,’ ‘);
const m = t.match(/(\d[\d\s.,]*\d|\d+)/);
if (m) {
const n = parseInt(m[0].replace(/[\s.,]/g,’’),10);
if (!isNaN(n) && n>500 && n<50000000) return { prix:n, prix_texte:t };
}
return { prix:null, prix_texte:t||null };
}
function devinerType(texte) {
const t=(texte||’’).toLowerCase();
if(/appartement|studio|flat|duplex|loft|penthouse/.test(t)) return ‘apartment’;
if(/maison|villa|bungalow|chalet|fermette|woning|huis/.test(t)) return ‘house’;
if(/immeuble|rapport|investissement|opbrengst/.test(t)) return ‘building’;
if(/terrain|parcelle|land|grond/.test(t)) return ‘land’;
if(/commerce|bureau|entrepôt|magasin|kantoor/.test(t)) return ‘commercial’;
return ‘other’;
}
function extraireVille(texte) {
if (!texte) return null;
const villes=[‘Liège’,‘Seraing’,‘Herstal’,‘Ans’,‘Grâce-Hollogne’,‘Oupeye’,‘Saint-Nicolas’,‘Beyne-Heusay’,‘Fléron’,‘Chaudfontaine’,‘Trooz’,‘Awans’,‘Juprelle’,‘Blegny’,‘Visé’,‘Flémalle’,‘Esneux’,‘Nandrin’,‘Neupré’,‘Soumagne’,‘Aywaille’,‘Sprimont’,‘Namur’,‘Jambes’,‘Dinant’,‘Andenne’,‘Gembloux’,‘Ciney’,‘Charleroi’,‘Mons’,‘Tournai’,‘La Louvière’,‘Mouscron’,‘Wavre’,‘Ottignies’,‘Nivelles’,‘Waterloo’,‘Bruxelles’,‘Ixelles’,‘Schaerbeek’,‘Anderlecht’,‘Molenbeek’,‘Uccle’,‘Etterbeek’,‘Woluwe’,‘Saint-Gilles’,‘Gand’,‘Anvers’,‘Bruges’,‘Louvain’,‘Mechelen’,‘Hasselt’,‘Kortrijk’,‘Roeselare’,‘Aalst’,‘Arlon’,‘Bastogne’];
for (const v of villes) {
if (texte.toLowerCase().includes(v.toLowerCase())) return v;
}
return null;
}
function normaliserUrl(href, agence) {
if (!href) return ‘’;
href=href.trim();
if (href.startsWith(‘http’)) return href;
if (href.startsWith(’//’)) return ‘https:’+href;
return href.startsWith(’/’) ? agence.site+href : agence.site+’/’+href;
}
async function fetchPage(url) {
try {
const res = await fetch(url, {
headers:{‘User-Agent’:‘Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/124.0.0 Safari/537.36’,‘Accept’:’text/html,application/xhtml+xml;q=0.9,*/*;q=0.8’,‘Accept-Language’:‘fr-BE,fr;q=0.9,nl;q=0.7’},
signal: AbortSignal.timeout(10000),
});
if (!res.ok) { console.log(`  ⚠️  HTTP ${res.status} — ${url}`); return null; }
return res.text();
} catch(e) { console.log(`  ⚠️  ${e.message} — ${url}`); return null; }
}

// ── Scraper une agence ────────────────────────────────────────────

async function scraperAgence(agence, sb) {
let nouvelles = 0;
for (const page of agence.pages) {
console.log(`  → ${page.url}`);
const html = await fetchPage(page.url);
if (!html) continue;
const $ = cheerio.load(html);
const elements = $(agence.sel.conteneur);
console.log(`     ${elements.length} annonce(s) détectée(s)`);
if (elements.length === 0) continue;
for (let i = 0; i < elements.length; i++) {
const el = elements.eq(i);
const lienBrut = el.find(agence.sel.lien).first().attr(‘href’) || el.closest(‘a’).attr(‘href’) || ‘’;
const url = normaliserUrl(lienBrut, agence);
if (!url || !url.startsWith(‘http’)) continue;
const urlHash = hash(url);
const { data: existant } = await sb.from(‘listings’).select(‘id’).eq(‘url_hash’, urlHash).maybeSingle();
if (existant) continue;
const titre    = el.find(agence.sel.titre).first().text().trim() || ‘Annonce sans titre’;
const adresse  = agence.sel.adresse ? el.find(agence.sel.adresse).first().text().trim() : null;
const imgEl    = el.find(agence.sel.photo).first();
const photoSrc = imgEl.attr(‘data-src’) || imgEl.attr(‘data-lazy-src’) || imgEl.attr(‘src’) || ‘’;
const photoUrl = photoSrc && !photoSrc.startsWith(‘data:’) ? normaliserUrl(photoSrc, agence) : null;
const { prix, prix_texte } = extrairePrix(el.find(agence.sel.prix).first().text());
const { error } = await sb.from(‘listings’).insert({
agency_id:agence.id, agency_name:agence.nom, agency_url:agence.site,
title:titre, address:adresse||null, city:extraireVille(adresse||titre),
price:prix, price_text:prix_texte, type:devinerType(titre),
status:page.statut, photos:photoUrl?[photoUrl]:[], url, url_hash:urlHash, is_new:true,
});
if (!error) { nouvelles++; console.log(`     ✅ ${titre.slice(0,55)}`); }
}
await new Promise(r => setTimeout(r, 1500));
}
return nouvelles;
}

// ── Notifications email ───────────────────────────────────────────

async function envoyerNotifications(sb) {
const resendKey = process.env.RESEND_API_KEY;
if (!resendKey) return;
const { data: nouvelles } = await sb.from(‘listings’).select(’*’).eq(‘is_new’,true).limit(20);
if (!nouvelles || nouvelles.length === 0) return;
const { data: abonnes } = await sb.from(‘subscribers’).select(’*’).eq(‘active’,true);
if (!abonnes || abonnes.length === 0) return;
const appUrl = process.env.VERCEL_URL ? `https://${process.env.VERCEL_URL}` : ‘https://immo-radar-belgique.vercel.app’;
const fmtPrix = l => l.price ? new Intl.NumberFormat(‘fr-BE’,{style:‘currency’,currency:‘EUR’,maximumFractionDigits:0}).format(l.price) : (l.price_text||‘Prix sur demande’);
for (const sub of abonnes) {
const f = sub.filters||{};
const filtrées = nouvelles.filter(l => {
if (f.status && f.status!==l.status) return false;
if (f.type   && f.type!==l.type)     return false;
if (f.city   && l.city && !l.city.toLowerCase().includes(f.city.toLowerCase())) return false;
if (f.maxPrice && l.price && l.price>f.maxPrice) return false;
return true;
});
if (filtrées.length===0) continue;
const cartes = filtrées.slice(0,5).map(l=>` <div style="border:1px solid #2a3a55;border-radius:10px;margin-bottom:16px;overflow:hidden;background:#1a2436;"> ${l.photos&&l.photos[0]?`<img src="${l.photos[0]}" style="width:100%;height:150px;object-fit:cover;" alt="">`:''} <div style="padding:14px;"> <p style="margin:0 0 4px;font-size:11px;color:#c9973a;font-weight:600;text-transform:uppercase;">${l.agency_name}</p> <p style="margin:0 0 8px;font-size:15px;font-weight:600;color:#f5f0e8;">${l.title}</p> <p style="margin:0 0 10px;font-size:20px;font-weight:700;color:#e8b654;">${fmtPrix(l)}</p> ${l.address?`<p style="margin:0 0 12px;font-size:12px;color:#8a9ab5;">📍 ${l.address}</p>`:''} <a href="${l.url}" style="display:inline-block;background:#c9973a;color:#0f1724;padding:8px 18px;border-radius:6px;text-decoration:none;font-size:12px;font-weight:700;">Voir →</a> </div> </div>`).join(’’);
await fetch(‘https://api.resend.com/emails’,{
method:‘POST’,
headers:{‘Authorization’:`Bearer ${resendKey}`,‘Content-Type’:‘application/json’},
body:JSON.stringify({
from: process.env.RESEND_FROM||‘Immo Radar [noreply@immo-radar.vercel.app](mailto:noreply@immo-radar.vercel.app)’,
to: sub.email,
subject: `🏠 ${filtrées.length} nouvelle(s) annonce(s) en Belgique`,
html:`<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head><body style="background:#0f1724;color:#f5f0e8;font-family:sans-serif;padding:24px;"><div style="max-width:560px;margin:0 auto;"><h1 style="color:#e8b654;font-size:20px;text-align:center;margin-bottom:20px;">🏠 Immo Radar Belgique</h1>${cartes}<div style="text-align:center;margin-top:16px;"><a href="${appUrl}" style="background:#1a2436;border:1px solid #c9973a;color:#c9973a;padding:10px 24px;border-radius:8px;text-decoration:none;font-size:13px;">Toutes les annonces</a></div><p style="text-align:center;font-size:11px;color:#4a5a75;margin-top:20px;"><a href="${appUrl}/api/scrape?unsub=${encodeURIComponent(sub.email)}" style="color:#4a5a75;">Se désabonner</a></p></div></body></html>`,
}),
});
}
await sb.from(‘listings’).update({is_new:false}).eq(‘is_new’,true);
}

// ── Point d’entrée Vercel ─────────────────────────────────────────

module.exports = async function handler(req, res) {
if (req.method===‘GET’ && req.query.unsub) {
const sb = getSupabase();
await sb.from(‘subscribers’).update({active:false}).eq(‘email’,req.query.unsub);
return res.redirect(302,’/’);
}
const secret = process.env.CRON_SECRET;
if (secret) {
const auth = req.headers[‘authorization’]||’’;
if (auth!==`Bearer ${secret}`) return res.status(401).json({erreur:‘Non autorisé’});
}
console.log(`\n${'═'.repeat(55)}\n  Immo Radar Belgique — ${new Date().toLocaleString('fr-BE')}\n  ${AGENCES.length} agences à scraper\n${'═'.repeat(55)}`);
let sb;
try { sb = getSupabase(); } catch(e) { return res.status(500).json({erreur:e.message}); }
const resultats = [];
let totalNouvelles = 0;
for (const agence of AGENCES) {
console.log(`\n📡 ${agence.nom}`);
try {
const n = await scraperAgence(agence, sb);
resultats.push({agence:agence.nom, nouvelles:n});
totalNouvelles += n;
} catch(e) {
console.log(`  ❌ ${e.message}`);
resultats.push({agence:agence.nom, nouvelles:0, erreur:e.message});
}
}
if (totalNouvelles > 0) {
console.log(`\n📧 ${totalNouvelles} nouvelles → notifications…`);
try { await envoyerNotifications(sb); } catch(e) { console.log(‘⚠️’,e.message); }
}
console.log(`\n✅ Terminé — ${totalNouvelles} nouvelle(s) annonce(s)\n`);
return res.status(200).json({ok:true, timestamp:new Date().toISOString(), totalNouvelles, resultats});
};
